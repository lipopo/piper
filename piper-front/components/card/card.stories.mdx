### TrendingCard

流行面板展示卡片

<Preview>
    <Story
        name='trending-card'
        args={{
            flowName: 'Thrending Flow'
        }}
    >
        {((args, { argTypes }) => (
            {
                props: ['flowName'],
                components: { TrendingCard },
                template: `
                    <trending-card
                        :flowName='flowName'>
                    </trending-card>`
            }
        )).bind({})}
    </Story>
</Preview>

### FlowCard

展示已经创建的流的卡片

<Preview>
    <Story
        name='flow-card'
        args={{
            flowName: 'BaseName',
            imgPath: '',
            updateTime: new Date(),
            now: new Date()
        }}
        argTypes={{
            now: {
                control: 'date'
            },
            updateTime: {
                control: 'date'
            }
        }}
    >{((args, { argTypes }) => ({
        components: { FlowCard },
        props: ['flowName', 'imgPath', 'updateTime', 'now'],
        template: `<flow-card :flowName='flowName' :imgPath='imgPath' :updateTime='updateTime' :now='now'></flow-card>`
    })).bind({})}</Story>
</Preview>
